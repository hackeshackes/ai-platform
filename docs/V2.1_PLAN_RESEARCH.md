### 4.2 技术决策

#### Feature Store设计

```python
# backend/feature_store/store.py
class FeatureStore:
    """
    特征存储服务
    
    对标: MLflow, Feast
    """
    
    async def create_feature_group(
        name: str,
        schema: List[Feature],
        source: str
    ) -> FeatureGroup:
        """创建特征组"""
        
    async def write_features(
        feature_group: str,
        dataframe: DataFrame,
        mode: str = "upsert"
    ) -> str:
        """写入特征"""
        
    async def get_feature_vector(
        entity_id: str,
        features: List[str]
    ) -> Dict:
        """获取特征向量"""
```

#### Model Registry设计

```python
# backend/models/registry.py
class ModelRegistry:
    """
    模型注册中心
    
    对标: MLflow Model Registry
    """
    
    async def register_model(
        name: str,
        version: str,
        model_uri: str,
        description: str = ""
    ) -> ModelVersion:
        """注册模型"""
        
    async def transition_stage(
        model_name: str,
        version: str,
        stage: str  # None, Staging, Production, Archived
    ) -> ModelVersion:
        """切换阶段"""
        
    async def get_latest_versions(model_name: str) -> List[ModelVersion]:
        """获取最新版本"""
```
